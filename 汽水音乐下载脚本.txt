    const CONFIG = {
        maxSearchTime: 30000,
        searchInterval: 500,
        debug: true

    let audioInfo = {
        url: null,
        title: '未知歌曲',
        artist: '未知艺术家',
        cover: '',
        found: false
    };

    function log(...args) {
        if (CONFIG.debug) {
            console.log('📢 [汽水音乐]', ...args);
        }
    }

    function isValidMediaUrl(url) {
        if (!url) return false;
        const lowerUrl = url.toLowerCase();
        if (lowerUrl.includes('logo.png') || lowerUrl.includes('favicon') || lowerUrl.includes('.png') || lowerUrl.includes('.jpg') || lowerUrl.includes('.jpeg') || lowerUrl.includes('.gif') || lowerUrl.includes('.webp')) {
            return false;
        }
        if (!lowerUrl.includes('douyinvod.com')) return false;
        if (lowerUrl.includes('/video/') || lowerUrl.includes('/audio/') || lowerUrl.includes('mime_type=audio_mp4') || lowerUrl.includes('mime_type=video_mp4') || lowerUrl.endsWith('.mp3') || lowerUrl.endsWith('.mp4') || lowerUrl.endsWith('.m4a')) {
            return true;
        }
        return false;
    }

    function isInvalidCover(url) {
        if (!url) return true;
        const lower = url.toLowerCase();
        const blacklist = ['logo', 'favicon', 'icon', 'placeholder', 'default'];
        for (let word of blacklist) {
            if (lower.includes(word)) return true;
        }
        return false;
    }

    function extractCoverFromPage() {
        let coverUrl = '';
        const metaOgImage = document.querySelector('meta[property="og:image"]');
        if (metaOgImage) {
            const content = metaOgImage.getAttribute('content');
            if (content && !isInvalidCover(content)) coverUrl = content;
        }
        if (!coverUrl) {
            const metaTwitterImage = document.querySelector('meta[name="twitter:image"]');
            if (metaTwitterImage) {
                const content = metaTwitterImage.getAttribute('content');
                if (content && !isInvalidCover(content)) coverUrl = content;
            }
        }
        if (!coverUrl) {
            const images = document.querySelectorAll('img');
            for (let img of images) {
                if (img.src && (img.src.includes('douyinpic.com') || img.src.includes('qishui'))) {
                    if (!isInvalidCover(img.src)) {
                        coverUrl = img.src;
                        break;
                    }
                }
            }
        }
        if (!coverUrl && window._ROUTER_DATA) {
            try {
                const trackData = window._ROUTER_DATA?.loaderData?.track_page?.audioWithLyricsOption;
                if (trackData?.coverURL && !isInvalidCover(trackData.coverURL)) coverUrl = trackData.coverURL;
                const videoData = window._ROUTER_DATA?.loaderData?.ugc_video_page?.videoOptions;
                if (videoData?.coverURL && !isInvalidCover(videoData.coverURL)) coverUrl = videoData.coverURL;
                else if (videoData?.firstFrameURL && !isInvalidCover(videoData.firstFrameURL)) coverUrl = videoData.firstFrameURL;
            } catch(e) {}
        }
        return coverUrl;
    }

    function extractFromRouterData() {
        try {
            if (!window._ROUTER_DATA) return null;
            const trackPage = window._ROUTER_DATA?.loaderData?.track_page;
            if (trackPage?.audioWithLyricsOption?.url) {
                const opt = trackPage.audioWithLyricsOption;
                return {
                    url: opt.url,
                    title: opt.trackName || '未知歌曲',
                    artist: opt.artistName || '未知艺术家',
                    cover: opt.coverURL || '',
                    source: 'track_page'
                };
            }
            const ugcPage = window._ROUTER_DATA?.loaderData?.ugc_video_page;
            if (ugcPage?.videoOptions?.url) {
                const opt = ugcPage.videoOptions;
                return {
                    url: opt.url,
                    title: opt.videoName || '未知歌曲',
                    artist: opt.artistName || '未知艺术家',
                    cover: opt.coverURL || opt.firstFrameURL || '',
                    source: 'ugc_video_page'
                };
            }
        } catch(e) {
            log('解析_ROUTER_DATA失败', e);
        }
        return null;
    }

    function searchAudioInPage() {
        log('屎山代码正在发力🤔');

        const routerResult = extractFromRouterData();
        if (routerResult && isValidMediaUrl(routerResult.url)) {
            audioInfo = {
                url: routerResult.url,
                title: routerResult.title,
                artist: routerResult.artist,
                cover: routerResult.cover || extractCoverFromPage(),
                found: true
            };
            log('屎山代码匹配成功😋(路由数据):', audioInfo);
            return true;
        }

        const results = [];
        let extractedCover = extractCoverFromPage();

        const scripts = document.querySelectorAll('script');
        scripts.forEach((script, index) => {
            const content = script.textContent || script.innerText;
            if (!content) return;

            if (content.includes('_ROUTER_DATA')) {
                log(`找到 _ROUTER_DATA 脚本 #${index}`);
                const routerMatch = content.match(/_ROUTER_DATA\s*=\s*({.*?});/s);
                if (routerMatch) {
                    try {
                        const data = JSON.parse(routerMatch[1]);
                        const trackData = data?.loaderData?.track_page?.audioWithLyricsOption;
                        if (trackData?.url && isValidMediaUrl(trackData.url)) {
                            results.push({
                                url: trackData.url,
                                title: trackData.trackName,
                                artist: trackData.artistName,
                                cover: trackData.coverURL,
                                source: '_ROUTER_DATA JSON'
                            });
                        }
                        const videoData = data?.loaderData?.ugc_video_page?.videoOptions;
                        if (videoData?.url && isValidMediaUrl(videoData.url)) {
                            results.push({
                                url: videoData.url,
                                title: videoData.videoName || videoData.artistName,
                                artist: videoData.artistName,
                                cover: videoData.coverURL,
                                source: 'ugc_video JSON'
                            });
                        }
                    } catch (e) {}
                }

                const urlMatches = content.matchAll(/"url"\s*:\s*"([^"]+)/g);
                for (const match of urlMatches) {
                    let url = match[1].replace(/\\u002F/g, '/');
                    if (isValidMediaUrl(url)) {
                        let title = '未知歌曲';
                        const titleMatch = content.match(/"trackName"\s*:\s*"([^"]+)/);
                        if (titleMatch) title = titleMatch[1];
                        else {
                            const vnameMatch = content.match(/"videoName"\s*:\s*"([^"]+)/);
                            if (vnameMatch) title = vnameMatch[1];
                        }
                        let artist = '未知作者';
                        const artistMatch = content.match(/"artistName"\s*:\s*"([^"]+)/);
                        if (artistMatch) artist = artistMatch[1];
                        results.push({
                            url: url,
                            title: title,
                            artist: artist,
                            source: '正则提取'
                        });
                    }
                }
            }

            if (content.includes('douyinvod.com') || content.includes('audio/mp4') || content.includes('v5-luna')) {
                log(`屎山代码成功发力😋找到包含音视频的部分 #${index}`);
                const urlMatches = content.matchAll(/(https?:\\?\/\\?\/[^"'\s]+(?:douyinvod\.com|v5-luna\.douyinvod\.com)[^"'\s]+)/g);
                for (const match of urlMatches) {
                    let url = match[1].replace(/\\u002F/g, '/').replace(/\\\//g, '/');
                    if (isValidMediaUrl(url)) {
                        results.push({
                            url: url,
                            title: '未知歌曲',
                            artist: '未知作者',
                            source: '脚本URL匹配'
                        });
                    }
                }
            }
        });

        const audioEls = document.querySelectorAll('audio');
        audioEls.forEach((audio, index) => {
            if (audio.src && isValidMediaUrl(audio.src)) {
                results.push({
                    url: audio.src,
                    title: audio.title || '未知歌曲',
                    artist: '未知作者',
                    source: `audio标签 #${index}`
                });
            }
        });

        const videoEls = document.querySelectorAll('video');
        videoEls.forEach((video, index) => {
            if (video.src && isValidMediaUrl(video.src)) {
                results.push({
                    url: video.src,
                    title: video.title || '未知歌曲',
                    artist: '未知作者',
                    source: `video标签 #${index}`
                });
                if (!extractedCover && video.poster && !isInvalidCover(video.poster)) extractedCover = video.poster;
            }
        });

        const elementsWithSrc = document.querySelectorAll('[src]');
        elementsWithSrc.forEach(el => {
            const src = el.getAttribute('src');
            if (src && isValidMediaUrl(src)) {
                results.push({
                    url: src,
                    title: el.title || '未知歌曲',
                    artist: '未知作者',
                    source: '元素src属性'
                });
            }
        });

        const allElements = document.querySelectorAll('*');
        allElements.forEach(el => {
            const attributes = el.attributes;
            for (let attr of attributes) {
                if (attr.name.startsWith('data-') && attr.value && isValidMediaUrl(attr.value)) {
                    results.push({
                        url: attr.value,
                        title: '未知歌曲',
                        artist: '未知作者',
                        source: `data属性: ${attr.name}`
                    });
                }
            }
        });

        if (window.performance && window.performance.getEntries) {
            const entries = window.performance.getEntries();
            entries.forEach(entry => {
                if (entry.name && isValidMediaUrl(entry.name)) {
                    results.push({
                        url: entry.name,
                        title: '未知歌曲',
                        artist: '未知作者',
                        source: '网络请求'
                    });
                }
            });
        }

        const uniqueResults = [];
        const seenUrls = new Set();
        results.forEach(item => {
            if (item.url && !seenUrls.has(item.url)) {
                seenUrls.add(item.url);
                uniqueResults.push(item);
            }
        });

        if (uniqueResults.length > 0) {
            log(`找到 ${uniqueResults.length} 个唯一音视频链接`);
            const bestMatch = uniqueResults.find(r => r.title !== '未知歌曲') || uniqueResults[0];
            audioInfo = {
                url: bestMatch.url,
                title: bestMatch.title,
                artist: bestMatch.artist,
                cover: extractedCover,
                found: true
            };
            log('屎山代码匹配成功😋:', audioInfo);
            return true;
        }

        log('屎山代码炸了😨啥也没找到🧐');
        return false;
    }

        });

        downloadBtn.onclick = async () => {
            const originalText = downloadBtn.innerHTML;
            downloadBtn.innerHTML = '⏳';
            downloadBtn.disabled = true;

            try {
                const filename = `${audioInfo.title}_${audioInfo.artist}-ZynSync提供下载.mp3`
                    .replace(/[\\/*?:"<>|]/g, '_')
                    .substring(0, 100);

                let downloaded = false;

                if (typeof GM_download !== 'undefined') {
                    try {
                        await new Promise((resolve, reject) => {
                            GM_download({
                                url: audioInfo.url,
                                name: filename,
                                onload: resolve,
                                onerror: reject,
                                timeout: 30000
                            });
                        });
                        downloaded = true;
                    } catch (e) {
                        log('GM_download失败:', e);
                    }
                }

                if (!downloaded && typeof GM_xmlhttpRequest !== 'undefined') {
                    try {
                        await new Promise((resolve, reject) => {
                            GM_xmlhttpRequest({
                                method: 'GET',
                                url: audioInfo.url,
                                responseType: 'blob',
                                onload: (resp) => {
                                    const blob = resp.response;
                                    const blobUrl = URL.createObjectURL(blob);
                                    const a = document.createElement('a');
                                    a.href = blobUrl;
                                    a.download = filename;
                                    a.click();
                                    URL.revokeObjectURL(blobUrl);
                                    resolve();
                                },
                                onerror: reject
                            });
                        });
                        downloaded = true;
                    } catch (e) {
                        log('GM_xmlhttpRequest失败:', e);
                    }
                }

                if (!downloaded) {
                    try {
                        const response = await fetch(audioInfo.url, {
                            mode: 'cors',
                            credentials: 'omit'
                        });
                        const blob = await response.blob();
                        const blobUrl = URL.createObjectURL(blob);
                        const a = document.createElement('a');
                        a.href = blobUrl;
                        a.download = filename;
                        a.click();
                        URL.revokeObjectURL(blobUrl);
                        downloaded = true;
                    } catch (e) {
                        log('fetch失败:', e);
                    }
                }

                if (!downloaded) {
                    window.open(audioInfo.url, '_blank');
                    downloaded = true;
                }

                downloadBtn.innerHTML = '✅';
                setTimeout(() => {
                    downloadBtn.innerHTML = originalText;
                    downloadBtn.disabled = false;
                }, 1500);

            } catch (error) {
                console.error('下载失败:', error);
                downloadBtn.innerHTML = '❌';
                setTimeout(() => {
                    downloadBtn.innerHTML = originalText;
                    downloadBtn.disabled = false;
                }, 1500);
            }
        };

        copyBtn.onclick = () => {
            navigator.clipboard.writeText(audioInfo.url).then(() => {
                const originalText = copyBtn.innerHTML;
                copyBtn.innerHTML = '✅';
                setTimeout(() => {
                    copyBtn.innerHTML = originalText;
                }, 1000);
            }).catch(() => {
                prompt('链接:', audioInfo.url);
            });
        };

        beautyBtn.onclick = () => {
            replaceWithBeautyPlayer();
        };

        infoCard.onclick = () => {
            alert(`歌曲：${audioInfo.title}\n作者：${audioInfo.artist}`);
        };

        buttonGroup.appendChild(copyBtn);
        buttonGroup.appendChild(beautyBtn);
        container.appendChild(infoCard);
        container.appendChild(buttonGroup);
        document.body.appendChild(container);

        if (typeof GM_notification !== 'undefined') {
            GM_notification({
                text: `找到: ${audioInfo.title}`,
                title: '汽水音乐',
                timeout: 2000
            });
        }
    }

    function generateDefaultCover(songName = '') {
        const canvas = document.createElement('canvas');
        canvas.width = 500;
        canvas.height = 500;

        ctx.strokeStyle = '#ffffff';
        ctx.lineWidth = 3;

    async function loadAudioFromFile(file) {
        if (!file) return false;
        if (audio.src && audio.src.startsWith('blob:')) {

    async function initAudioContextAndAnalyser() {
        if (!audioContext) {
            audioContext = new (window.AudioContext || window.webkitAudioContext)();
        }
        if (audioContext.state === 'suspended') {
            await audioContext.resume();
        }
        if (sourceNode) {
            try { sourceNode.disconnect(); } catch(e) {}
        }
        if (analyserNode) {
            try { analyserNode.disconnect(); } catch(e) {}
        }
        if (audio.src) {
            try {
                sourceNode = audioContext.createMediaElementSource(audio);
                analyserNode = audioContext.createAnalyser();
                analyserNode.fftSize = 256;
                sourceNode.connect(analyserNode);
                analyserNode.connect(audioContext.destination);
                isAudioContextReady = true;
                startBeatAnimation();
            } catch(e) {
                console.warn("无法建立音频分析节点，律动效果可能失效", e);
                isAudioContextReady = false;
            }
        }
    }

    function startBeatAnimation() {
        if (animationFrameId) cancelAnimationFrame(animationFrameId);
        function animateBeat() {
            if (!analyserNode || !isAudioContextReady || !audio.src || audio.paused) {
                smoothScale = 1;
                smoothRotate = 0;
                filteredRotateAmp = 0;
                artworkFrame.style.transform = 'scale(1) rotate(0deg)';
                animationFrameId = requestAnimationFrame(animateBeat);
                return;
            }
            const dataArray = new Uint8Array(analyserNode.frequencyBinCount);
            analyserNode.getByteTimeDomainData(dataArray);
            let sum = 0;
            for (let i = 0; i < dataArray.length; i++) {
                let v = (dataArray[i] - 128) / 128;
                sum += v * v;
            }
            let avg = Math.sqrt(sum / dataArray.length);
            let intensity = Math.min(0.45, avg * 1.8);
            let targetScale = 1 + intensity * 0.55;
            let targetRotateAmp = intensity * 7.5;
            filteredRotateAmp = filteredRotateAmp * 0.92 + targetRotateAmp * 0.08;
            smoothScale = smoothScale * 0.88 + targetScale * 0.12;
            phase += 0.18;
            let rotateVal = Math.sin(phase) * filteredRotateAmp;
            smoothRotate = smoothRotate * 0.86 + rotateVal * 0.14;
            artworkFrame.style.transform = 'scale(' + smoothScale + ') rotate(' + smoothRotate + 'deg)';
            animationFrameId = requestAnimationFrame(animateBeat);
        }
        animateBeat();
    }

    async function loadAndPlayTrack(index) {
        if (index < 0 || index >= currentTrackList.length) return;
        const fileOrUrl = currentTrackList[index];
        if (typeof fileOrUrl === 'string') {
            if (audio.src && audio.src.startsWith('blob:')) {
                URL.revokeObjectURL(audio.src);
            }
            audio.crossOrigin = 'anonymous';
            audio.src = fileOrUrl;
            await new Promise((resolve) => {
                audio.addEventListener('loadedmetadata', () => {
                    if (currentTrackList[index].title) {
                        trackTitleEl.textContent = currentTrackList[index].title;
                        trackArtistEl.textContent = currentTrackList[index].artist;
                    }
                    if (currentTrackList[index].cover && currentTrackList[index].cover !== '') {
                        coverImg.src = currentTrackList[index].cover;
                    } else {
                        coverImg.src = generateDefaultCover(currentTrackList[index].title || '歌曲');
                    }
                    durationTimeEl.textContent = formatTime(audio.duration);
                    updateProgress();
                    resolve();
                }, { once: true });
                audio.load();
            });
        } else {
            await loadAudioFromFile(fileOrUrl);
        }
        if (audioContext && audioContext.state === 'suspended') await audioContext.resume();
        await audio.play().catch(e => console.log("播放:", e));

        filename = filename.replace(/[\\\\/*?:"<>|]/g, '_');
        try {
            const response = await fetch(downloadUrl, { mode: 'cors', credentials: 'omit' });
            if (!response.ok) throw new Error('下载失败');
            const blob = await response.blob();
            const blobUrl = URL.createObjectURL(blob);
            const a = document.createElement('a');
            a.href = blobUrl;
            a.download = filename;

            a.click();

            window.open(downloadUrl, '_blank');
        }

            if (searchAudioInPage()) {
                log(`✅ 在第 ${searchCount} 次搜索成功！`);
                clearInterval(searchInterval);
                createDownloadButton();
            } else if (elapsed > CONFIG.maxSearchTime) {
                log(`❌ 搜索超时 (${CONFIG.maxSearchTime}ms)`);
                clearInterval(searchInterval);
                createManualSearchButton();
